### Define the function of constructing dataset according the conditions
def construct_dataset(a, b):
    df = pd.DataFrame(data)
    df.columns = ['z_i1', 'z_i2', 'z_i3']

    y_i1 = 1 + df['z_i1']
    y_i2 = 5 + 2 * df['z_i1'] + df['z_i2']
    u_i = a * (y_i1 - 1) + b * (y_i2 - 5) + df['z_i3']

    m_i2 = []
    for i in range(rows):
      if u_i[i] < 0:
        m_i2.append(1)
      else:
        m_i2.append(0)

    y_i2_miss = []
    for i in range(rows):
      if m_i2[i] == 1:
        y_i2_miss.append(np.nan)
      else:
        y_i2_miss.append(y_i2[i])

    df['y_i1'] = y_i1
    df['y_i2'] = y_i2
    df['u_i'] = u_i
    df['m_i2'] = m_i2
    df['y_i2_miss'] = y_i2_miss

    return(df)
